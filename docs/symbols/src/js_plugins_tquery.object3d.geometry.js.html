 11 tQuery.Object3D.register('scale', function(scale){
 12 	// handle parameters
 13 	if( typeof scale === "number" && arguments.length === 1 ){
 14 		scale	= new THREE.Vector3(scale, scale, scale);
 15 	}else if( typeof scale === "number" && arguments.length === 3 ){
 16 		scale	= new THREE.Vector3(arguments[0], arguments[1], arguments[2]);
 17 	}
 18 	console.assert(scale instanceof THREE.Vector3, "Geometry.scale parameter error");
 19 
 20 	// do the operation on each node
 21 	this.each(function(object3d){
 22 		object3d.scale.copy(scale);
 23 	});
 24 
 25 	// return this, to get chained API	
 26 	return this;
 27 });
