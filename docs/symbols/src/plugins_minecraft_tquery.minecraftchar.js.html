 16 	function cubeFromPlanes(size, material) {
 17 		var cube	= new THREE.Object3D();
 18 		var meshes	= [];
 19 		for( var i=0; i < 6; i++ ){
 20 			var geometry	= new THREE.PlaneGeometry(size, size);
 21 			var mesh	= new THREE.Mesh(geometry, material);
 22 			mesh.doubleSided= true;
 23 			//mesh.visible	= false
 24 			cube.add(mesh);
 25 			meshes.push(mesh);
 26 		}
 27 		// Front
 28 		meshes[0].rotation.x	=  Math.PI/2;
 29 		meshes[0].position.z	=  size/2;
 30 
 31 		// Back
 32 		meshes[1].rotation.x	=  Math.PI/2;
 33 		meshes[1].rotation.z	=  Math.PI;
 34 		meshes[1].position.z	= -size/2;
 35 
 36 		// Top
 37 		meshes[1].rotation.y	=  Math.PI/2;
 38 		meshes[2].position.y	=  size/2;
 39 
 40 		// Bottom
 41 		meshes[3].rotation.y	= -Math.PI/2;
 42 		meshes[3].rotation.z	=  Math.PI;
 43 		meshes[3].position.y	= -size/2;
 44 
 45 		// Left
 46 		meshes[4].rotation.x	=  Math.PI/2;
 47 		meshes[4].rotation.z	=  Math.PI/2;
 48 		meshes[4].position.x	= -size/2;
 49 
 50 		// Right
 51 		meshes[5].rotation.x	= -Math.PI/2;
 52 		meshes[5].rotation.y	=  Math.PI;
 53 		meshes[5].rotation.z	=  Math.PI/2;
 54 		meshes[5].position.x	=  size/2;
 55 
 56 		return cube;
 57 	};
 58 	function getMaterial(image, transparent) {
 59 		var texture		= new THREE.Texture(image);
 60 		texture.magFilter	= THREE.NearestFilter;
 61 		texture.minFilter	= THREE.NearestFilter;
 62 		texture.format		= transparent ? THREE.RGBAFormat : THREE.RGBFormat;
 63 		texture.needsUpdate	= true;
 64 		var material	= new THREE.MeshBasicMaterial({
 65 			map		: texture,
 66 			transparent	: transparent ? true : false
 67 		});
 68 		return material;
 69 	}
 70 	function uvmap (geometry, face, x, y, w, h, rotateBy) {
 71 		if( !rotateBy )	rotateBy = 0;
 72 		var uvs		= geometry.faceVertexUvs[0][face];
 73 		var tileU	= x;
 74 		var tileV	= y;
 75 		
 76 		uvs[ (0 + rotateBy) % 4 ].u = tileU * tileUvW;
 77 		uvs[ (0 + rotateBy) % 4 ].v = tileV * tileUvH;
 78 		uvs[ (1 + rotateBy) % 4 ].u = tileU * tileUvW;
 79 		uvs[ (1 + rotateBy) % 4 ].v = tileV * tileUvH	+ h * tileUvH;
 80 		uvs[ (2 + rotateBy) % 4 ].u = tileU * tileUvW	+ w * tileUvW;
 81 		uvs[ (2 + rotateBy) % 4 ].v = tileV * tileUvH	+ h * tileUvH;
 82 		uvs[ (3 + rotateBy) % 4 ].u = tileU * tileUvW	+ w * tileUvW;
 83 		uvs[ (3 + rotateBy) % 4 ].v = tileV * tileUvH;
 84 	};

292 /**
293  * Load a skin
294  *
295  * @param {string} url the url of the skin image
296 */
297 tQuery.MinecraftChar.prototype.loadSkin	= function(url){
298 	var canvas	= this._context.canvas;
299 	var context	= this._context;
300 	
301 	var image	= new Image();
302 	image.onload	= function () {
303 		context.clearRect(0, 0, canvas.width, canvas.height);
304 		context.drawImage(image, 0, 0);
305 
306 		this._material.map.needsUpdate		= true;
307 		this._materialTrans.map.needsUpdate	= true;
308 	}.bind(this);
309 	image.src = url;
310 	return this;	// for chained API
311 }
