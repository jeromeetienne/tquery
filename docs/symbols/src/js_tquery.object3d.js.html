1 /**
  2  * Handle object3D
  3  *
  4  * @class include THREE.Object3D
  5  *
  6  * @param {} object
  7  * @param {THREE.Object3D} rootnode
  8  * @returns {tQuery.*} the tQuery object created
  9 */
 10 tQuery.Object3D	= function(object, root){
 11 	if( typeof object === "string"){
 12 		object	= tQuery.Object3D._select(object, root);
 13 	}
 14 	this._lists	= object instanceof Array ? object : [object];
 15 	this.length	= this._lists.length;
 16 };

 27 /**
 28  * Retrieve the elements matched by the jQuery object
 29  * 
 30  * @param {Function} callback the function to notify. function(element){ }.
 31  * 			loop interrupted if it returns false
 32  * 
 33  * @returns {Boolean} return true if completed, false if interrupted
 34 */
 35 tQuery.Object3D.prototype.get	= function(idx){
 36 	if( idx === undefined )	return this._lists;
 37 
 38 	// sanity check - it MUST be defined
 39 	console.assert(this._lists[idx], "element not defined");
 40 	return this._lists[idx];
 41 };
 42 
 43 
 44 /**
 45  * loop over element
 46  * 
 47  * @param {Function} callback the function to notify. function(element){ }.
 48  * 			loop interrupted if it returns false
 49  * 
 50  * @returns {Boolean} return true if completed, false if interrupted
 51 */
 52 tQuery.Object3D.prototype.each	= function(callback){
 53 	for(var i = 0; i < this._lists.length; i++){
 54 		var object3d	= this._lists[i];
 55 		var keepLooping	= callback(object3d)
 56 		if( keepLooping === false )	return false;
 57 	}
 58 	return true;
 59 };

 65 /**
 66  * get geometry.
 67  * 
 68  * @returns {tQuery.Geometry} return the geometries from the tQuery.Object3D
 69 */
 70 tQuery.Object3D.prototype.geometry	= function(){
 71 	var geometries	= [];
 72 	this.each(function(object3d){
 73 		geometries.push(object3d.geometry)
 74 	});
 75 	return new tQuery.Geometry(geometries).back(this);

129 /**
130  * Getter/Setter for the id of the matched elements
131 */
132 tQuery.Object3D.prototype.id	= function(value)
133 {
134 	// sanity check 
135 	console.assert(this.length <= 1, "tQuery.Object3D.id used on multi-elements" );
136 	if( value !== undefined ){
137 		if( this.length > 0 ){
138 			var object3d	= this.get(0);
139 			object3d._tqId	= value;
140 		}
141 		return this;
142 	}else{
143 		if( this.length > 0 ){
144 			var object3d	= this.get(0);
145 			return object3d._tqId;
146 		}
147 		return undefined;
148 	}
149 };

180 /**
181  * return true if any of the matched elements has this class
182  * 
183  * @param {string} className the name of the class
184  * @returns {tQuery.Object3D} true if any of the matched elements has this class, false overwise
185 */
186 tQuery.Object3D.prototype.hasClass	= function(className){
187 	var completed	= this.each(function(object3d){
188 		// init ._tqClasses if needed
189 		object3d._tqClasses	= object3d._tqClasses	|| '';
190 
191 		var hasClass	= tQuery.Object3D._hasClassOne(object3d, className);
192 		return hasClass ? false : true;
193 	}.bind(this));
194 	return completed ? false : true;
195 };
196 
197 tQuery.Object3D._hasClassOne	= function(object3d, className){
198 	if( object3d._tqClasses === undefined )	return false;
199 	var classes	= object3d._tqClasses;
200 	var re		= new RegExp('(^| |\t)+('+className+')($| |\t)+');
201 	return classes.match(re) ? true : false;
202 };

209 tQuery.Object3D._select	= function(selector, root){
210 	root		= root	|| tQuery.scene.scene();	// FIXME scene is global
211 	var selectItems	= selector.split(' ').filter(function(v){ return v.length > 0;})
212 	var lists	= this._crawls(root, selectItems)
213 	return lists;
214 }
215 
216 tQuery.Object3D._crawls	= function(root, selectItems)
217 {
218 	var result	= [];
219 //console.log("crawl", root, selectItems)
220 	console.assert( selectItems.length >= 1 );
221 	var match	= this._selectItemMatch(root, selectItems[0]);
222 //console.log("  match", match)
223 	var nextSelect	= match ? selectItems.slice(1) : selectItems;
224 //console.log("  nextSelect", nextSelect)
225 
226 	if( nextSelect.length === 0 )	return [root];
227 
228 	root.children.forEach(function(child){
229 		var nodes	= this._crawls(child, nextSelect);
230 		// FIXME reallocate the array without need
231 		result		= result.concat(nodes);
232 	}.bind(this));
233 
234 	return result;
235 }
236 
237 tQuery.Object3D._selectItemMatch	= function(object3d, selectItem)
238 {
239 	// sanity check
240 	console.assert( object3d instanceof THREE.Object3D );
241 	console.assert( typeof selectItem === 'string' );
242 
243 	// all the geometries keywords
244 	// - Object.keys(THREE).filter(function(value){return value.match(/.+Geometry$/);}).map(function(value){ return value.replace(/Geometry$/,'').toLowerCase();});
245 	var geometries	= ["buffer", "cube", "cylinder", "extrude", "icosahedron", "lathe", "octahedron", "plane", "sphere", "text", "torus", "torusknot"];	
246 
247 	// parse selectItem into subItems
248 	var subItems	= selectItem.match(new RegExp("([^.#]+|\.[^.#]+|\#[^.#]+)", "g"));;
249 	// go thru each subItem
250 	var completed	= tQuery.each(subItems, function(subItem){
251 		var meta	= subItem.charAt(0);
252 		var suffix	= subItem.slice(1);
253 		//console.log("meta", meta, subItem, suffix, object3d)
254 		if( meta === "." ){
255 			var hasClass	= tQuery.Object3D._hasClassOne(object3d, suffix);
256 			return hasClass ? true : false;
257 		}else if( meta === "#" ){
258 			return object3d._tqId === suffix ? true : false;
259 		}else if( geometries.indexOf(subItem) !== -1 ){	// Handle geometries
260 			var geometry	= object3d.geometry;
261 			var className	= subItem.charAt(0).toUpperCase() + subItem.slice(1) + "Geometry";
262 			return geometry instanceof THREE[className];
263 		}else	console.assert(false, "invalid selector: "+subItem);
264 		return true;
265 	}.bind(this));
266 	
267 	return completed ? true : false;
268 }
