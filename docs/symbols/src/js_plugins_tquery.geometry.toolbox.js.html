 11 /**
 12  * zoom a geometry
 13  *
 14  * @name zoom
 15  * @methodOf tQuery.Geometry
 16 */
 17 tQuery.Geometry.register('zoom', function(vector3){
 18 	// handle parameters
 19 	if( typeof vector3 === "number" && arguments.length === 1 ){
 20 		vector3	= new THREE.Vector3(vector3, vector3, vector3);
 21 	}else if( typeof vector3 === "number" && arguments.length === 3 ){
 22 		vector3	= new THREE.Vector3(arguments[0], arguments[1], arguments[2]);
 23 	}
 24 	console.assert(vector3 instanceof THREE.Vector3, "Geometry.vector3 parameter error");
 25 
 26 	// change all geometry.vertices
 27 	this.each(function(geometry){
 28 		for(var i = 0; i < geometry.vertices.length; i++) {
 29 			var vertex	= geometry.vertices[i];
 30 			vertex.position.multiplySelf(vector3); 
 31 		}
 32 		// mark the vertices as dirty
 33 		geometry.__dirtyVertices = true;
 34 		geometry.computeBoundingBox();
 35 	})
 36 
 37 	// return this, to get chained API	
 38 	return this;
 39 });

 41 tQuery.Geometry.register('size', function(){
 42 	// only on zero-or-one element
 43 	console.assert(this.length <= 1)
 44 	// if no element, return undefined
 45 	if( this.length === 0 )	return undefined
 46 
 47 	// else measure the size of the element
 48 	var geometry	= this.get(0);
 49 	// compute middle
 50 	var size= new THREE.Vector3()
 51 	size.x	= geometry.boundingBox.max.x - geometry.boundingBox.min.x;
 52 	size.y	= geometry.boundingBox.max.y - geometry.boundingBox.min.y;
 53 	size.z	= geometry.boundingBox.max.z - geometry.boundingBox.min.z;
 54 
 55 	// return the just computed middle
 56 	return size;	
 57 });

 61 tQuery.Geometry.register('normalize', function(){
 62 	// change all geometry.vertices
 63 	this.each(function(geometry){
 64 		var node	= tQuery(geometry);
 65 		var size	= node.size();
 66 		if( size.x >= size.y && size.x >= size.z ){
 67 			node.scale(1/size.x);
 68 		}else if( size.y >= size.x && size.y >= size.z ){
 69 			node.scale(1/size.y);
 70 		}else{
 71 			node.scale(1/size.z);
 72 		}
 73 	});
 74 	// return this, to get chained API	
 75 	return this;
 76 });

 84 tQuery.Geometry.register('middlePoint', function(){
 85 	// only on zero-or-one element
 86 	console.assert(this.length <= 1)
 87 	// if no element, return undegined
 88 	if( this.length === 0 )	return undefined
 89 	// else measure the size of the element
 90 	var geometry	= this.get(0);
 91 	// compute middle
 92 	var middle	= new THREE.Vector3()
 93 	middle.x	= ( geometry.boundingBox.max.x + geometry.boundingBox.min.x ) / 2;
 94 	middle.y	= ( geometry.boundingBox.max.y + geometry.boundingBox.min.y ) / 2;
 95 	middle.z	= ( geometry.boundingBox.max.z + geometry.boundingBox.min.z ) / 2;
 96 
 97 	// return the just computed middle
 98 	return middle;
 99 });

105 tQuery.Geometry.register('translate', function(delta){
106 	// handle parameters
107 	if( typeof delta === "number" && arguments.length === 3 ){
108 		delta	= new THREE.Vector3(arguments[0], arguments[1], arguments[2]);
109 	}
110 	console.assert(delta instanceof THREE.Vector3, "Geometry.translate parameter error");
111 
112 	// change all geometry.vertices
113 	this.each(function(geometry){
114 		// change all geometry.vertices
115 		for(var i = 0; i < geometry.vertices.length; i++) {
116 			var vertex	= geometry.vertices[i];
117 			vertex.position.addSelf(delta); 
118 		}
119 		// mark the vertices as dirty
120 		geometry.__dirtyVertices = true;
121 		geometry.computeBoundingBox();
122 	})
123 
124 	// return this, to get chained API	
125 	return this;
126 });

128 tQuery.Geometry.register('rotate', function(angles, order){
129 	// handle parameters
130 	if( typeof angles === "number" && arguments.length === 3 ){
131 		angles	= new THREE.Vector3(arguments[0], arguments[1], arguments[2]);
132 	}
133 	console.assert(angles instanceof THREE.Vector3, "Geometry.rotate parameter error");
134 
135 	// set default rotation order if needed
136 	order	= order	|| 'XYZ';
137 	// compute transformation matrix
138 	var matrix	= new THREE.Matrix4();
139 	matrix.setRotationFromEuler(angles, order);
140 
141 	// change all geometry.vertices
142 	this.each(function(geometry){
143 		// apply the matrix
144 		geometry.applyMatrix( matrix );
145 	
146 		// mark the vertices as dirty
147 		geometry.__dirtyVertices = true;
148 		geometry.computeBoundingBox();
149 	});
150 
151 	// return this, to get chained API	
152 	return this;
153 });

157 tQuery.Geometry.register('center', function(noX, noY, noZ){
158 	// change all geometry.vertices
159 	this.each(function(geometry){
160 		// compute delta
161 		var delta	= this.middlePoint().negate();
162 		if( noX )	delta.x	= 0;
163 		if( noY )	delta.y	= 0;
164 		if( noZ )	delta.z	= 0;
165 
166 		return this.translate(delta)
167 	});
168 	// return this, to get chained API	
169 	return this;
170 });
