 16 /**
 17  * Create a cube
 18  * 
 19  * @returns {tQuery.Object3D} a tQuery.Object3D containing it
 20 */
 21 tQuery.register('cube', function(){
 22 	var material	= new THREE.MeshNormalMaterial();
 23 	var geometry	= new THREE.CubeGeometry(1,1,1);
 24 	geometry.dynamic	= true;
 25 	var mesh	= new THREE.Mesh(geometry, material)
 26 	return tQuery(mesh);
 27 });
