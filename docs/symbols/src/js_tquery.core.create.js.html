  5 /**

 12 /**
 13  * Create tQuery.loop
 14  * 
 15  * @param {THREE.Scene} scene the scene to display (optional)
 16  * @function
 17 */
 18 tQuery.register('createLoop', function(scene){
 19 	return new tQuery.Loop(scene);
 20 });
 21 
 22 /**
 23  * Create a cube
 24  * 
 25  * @returns {tQuery.Object3D} a tQuery.Object3D containing it
 26 */
 27 tQuery.register('createCube', function(){
 28 	var material	= new THREE.MeshNormalMaterial();
 29 	var geometry	= new THREE.CubeGeometry(1,1,1);
 30 	geometry.dynamic= true;
 31 	var mesh	= new THREE.Mesh(geometry, material)
 32 	return tQuery(mesh);
 33 });
 34 
 35 tQuery.register('createTorus', function(){
 36 	var material	= new THREE.MeshNormalMaterial();
 37 	var geometry	= new THREE.TorusGeometry( 0.5-0.15, 0.15 );
 38 	geometry.dynamic= true;
 39 	var mesh	= new THREE.Mesh(geometry, material)
 40 	return tQuery(mesh);
 41 });

 51 tQuery.register('createAxis', function(){
 52 	var axis	= new THREE.AxisHelper();
 53 	axis.scale.multiplyScalar(1/40);
 54 	return tQuery(axis);
 55 });
