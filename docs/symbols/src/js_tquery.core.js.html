  5 /**
  6  * Create a tQuery element
  7  *
  8  * @class root class
  9  * 
 10  * @param {} object
 11  * @param {THREE.Object3D} rootnode
 12  * @returns {tQuery.*} the tQuery object created
 13 */
 14 var tQuery	= function(object, root)
 15 {
 16 // TODO make tthat cleaner
 17 // - there is a list of functions registered by each plugins
 18 //   - handle() object instanceof THREE.Mesh
 19 //   - create() return new tQuery(object)
 20 // - this list is processed in order here
 21 
 22 	if( object instanceof THREE.Mesh  && tQuery.Mesh){
 23 		return new tQuery.Mesh(object);
 24 
 25 	}else if( object instanceof THREE.DirectionalLight && tQuery.DirectionalLight){
 26 		return new tQuery.DirectionalLight(object);
 27 	}else if( object instanceof THREE.AmbientLight && tQuery.AmbientLight){
 28 		return new tQuery.AmbientLight(object);
 29 	}else if( object instanceof THREE.Light && tQuery.Light){
 30 		return new tQuery.Light(object);
 31 
 32 	}else if( object instanceof THREE.Object3D  && tQuery.Object3D){
 33 		return new tQuery.Object3D(object);
 34 	}else if( object instanceof THREE.Geometry && tQuery.Geometry){
 35 		return new tQuery.Geometry(object);
 36 	}else if( object instanceof THREE.Material && tQuery.Material){
 37 		return new tQuery.Material(object);
 38 	}else if( typeof object === "string" && tQuery.Object3D){
 39 		return new tQuery.Object3D(object, root);
 40 	}else{
 41 		console.assert(false, "unsupported type")
 42 	}
 43 	return undefined;
 44 };

 55 /**
 56  * generic getter/setter
 57  * 
 58  * @param {Object} object the object in which store the data
 59  * @param {String} key the key/name of the data to get/set
 60  * @param {*} value the value to set (optional)
 61  * @param {Boolean} mustNotExist if true, ensure that the key doesnt already exist, optional default to false
 62  * 
 63  * @returns {*} return the value stored in this object for this key
 64 */
 65 tQuery.data	= function(object, key, value, mustNotExist)
 66 {
 67 	// sanity check
 68 	console.assert( object, 'invalid parameters' );
 69 	console.assert( typeof key === 'string', 'invalid parameters');
 70 
 71 	// init _tqData
 72 	object['_tqData']	= object['_tqData']	|| {};
 73 	// honor mustNotExist
 74 	if( mustNotExist ){
 75 		console.assert(object['_tqData'][key] === undefined, "This key already exists "+key);
 76 	}
 77 	// set the value if any
 78 	if( value ){
 79 		object['_tqData'][key]	= value;
 80 	}
 81 	// return the value
 82 	return object['_tqData'][key];
 83 };

 85 /**
 86  * Same as jQuery.removeData()
 87  *
 88  * @param {Boolean} mustExist if true, ensure the key does exist, default to false
 89 */
 90 tQuery.removeData	= function(object, key, mustExist)
 91 {
 92 	// handle the 'key as Array' case
 93 	if( key instanceof Array ){
 94 		key.forEach(function(key){
 95 			tQuery.removeData(object, key);
 96 		})
 97 		return;
 98 	}
 99 	// sanity check
100 	console.assert( typeof key === "string");
101 	// honor mustNotExist
102 	if( mustExist ){
103 		console.assert(object['_tqData'][key] !== undefined, "This key doesnt already exists "+key);
104 	}
105 	// do delete the key
106 	delete object['_tqData'][key];
107 	// TOTO remove object[_tqData] if empty now
108 }

115 /**
116  * loop over a Array.
117  * 
118  * @param {Array} arr the array to traverse.
119  * @param {Function} callback the function to notify. function(element){ }.
120  * 			loop interrupted if it returns false
121  * 
122  * @returns {Boolean} return true if completed, false if interrupted
123 */
124 tQuery.each	= function(arr, callback){
125 	for(var i = 0; i < arr.length; i++){
126 		var keepLooping	= callback(arr[i])
127 		if( keepLooping === false )	return false;
128 	}
129 	return true;
130 };

132 /**
133  * Make a child Class inherit from the parent class.
134  *
135  * @param {Object} childClass the child class which gonna inherit
136  * @param {Object} parentClass the class which gonna be inherited
137 */
138 tQuery.inherit	= function(childClass, parentClass){
139 	// trick to avoid calling parentClass constructor
140 	var tempFn		= function() {};
141 	tempFn.prototype	= parentClass.prototype;
142 	childClass.prototype	= new tempFn();
143 
144 	childClass.parent	= parentClass.prototype;
145 	childClass.prototype.constructor= childClass;	
146 };

151  * similar to jquery one but much smaller

153 tQuery.extend = function(obj, base){
154 	var result	= {};
155 	base && Object.keys(base).forEach(function(key){
156 		result[key]	= base[key];
157 	})
158 	obj && Object.keys(obj).forEach(function(key){
159 		result[key]	= obj[key];
160 	})
161 	return result;
162 };

168 /**
169  * Make an object pluginable
170  * 
171  * @param {Object} object the object on which you mixin function
172  * @param {Object} dest the object in which to register the plugin
173  * @param {string} suffix the suffix to add to the function name
174 */
175 tQuery._pluginsOn	= function(object, dest, fnNameSuffix){
176 	dest		= dest	|| object.prototype || object;
177 	fnNameSuffix	= fnNameSuffix || '';
178 	object['register'+fnNameSuffix]		= function(name, funct) {
179 		if( dest[name] ){
180 			throw new Error('Conflict! Already method called: ' + name);
181 		}
182 		dest[name]	= funct;
183 	};
184 	object['unregister'+fnNameSuffix]	= function(name){
185 		if( dest.hasOwnProperty(name) === false ){
186 			throw new Error('Plugin not found: ' + name);
187 		}
188 		delete dest[name];
189 	};
190 	object['registered'+fnNameSuffix]	= function(name){
191 		return dest.hasOwnProperty(name) === true;
192 	}
193 };
194 
195 tQuery.pluginsInstanceOn= function(klass){ return tQuery._pluginsOn(klass);			};
196 tQuery.pluginsStaticOn	= function(klass){ return tQuery._pluginsOn(klass, klass, 'Static');	};
197

199 tQuery.pluginsOn	= function(object, dest){
200 	console.warn("tQuery.pluginsOn is obsolete. prefere .pluginsInstanceOn, .pluginsStaticon");
201 	console.trace();
202 	return tQuery._pluginsOn(object, dest)

213 tQuery.mixinAttributes	= function(dstObject, properties){
214 	// mixin the new property
215 	// FIXME the inheritance should work now... not sure
216 	dstObject.prototype._attrProps	= tQuery.extend(dstObject.prototype._attrProps, properties);
217 
218 	dstObject.prototype.attr	= function(name, value){
219 		// handle parameters
220 		if( name instanceof Object && value === undefined ){
221 			Object.keys(name).forEach(function(key){
222 				this.attr(key, name[key]);
223 			}.bind(this));
224 		}else if( typeof(name) === 'string' ){
225 			console.assert( Object.keys(this._attrProps).indexOf(name) !== -1, 'invalid property name:'+name);
226 		}else	console.assert(false, 'invalid parameter');
227 
228 		// handle setter
229 		if( value !== undefined ){
230 			var convertFn	= this._attrProps[name];
231 			value		= convertFn(value);
232 			this.each(function(element){
233 				element[name]	= value;
234 			})
235 			return this;			
236 		}
237 		// handle getter
238 		if( this.length === 0 )	return undefined
239 		var element	= this.get(0);
240 		return element[name];
241 	};
242 
243 	// add shortcuts
244 	Object.keys(properties).forEach(function(name){
245 		dstObject.prototype[name]	= function(value){
246 			return this.attr(name, value);
247 		};
248 	}.bind(this));
249 };

255 /**
256  * Flow control - from https://github.com/jeromeetienne/gowiththeflow.js
257 */
258 tQuery.Flow	= function(){
259 	var self, stack = [], timerId = setTimeout(function(){ timerId = null; self._next(); }, 0);
260 	return self = {
261 		destroy	: function(){ timerId && clearTimeout(timerId);	},
262 		par	: function(callback, isSeq){
263 			if(isSeq || !(stack[stack.length-1] instanceof Array)) stack.push([]);
264 			stack[stack.length-1].push(callback);
265 			return self;
266 		},seq	: function(callback){ return self.par(callback, true);	},
267 		_next	: function(err, result){
268 			var errors = [], results = [], callbacks = stack.shift() || [], nbReturn = callbacks.length, isSeq = nbReturn == 1;
269 			callbacks && callbacks.forEach(function(fct, index){
270 				fct(function(error, result){
271 					errors[index]	= error;
272 					results[index]	= result;		
273 					if(--nbReturn == 0)	self._next(isSeq?errors[0]:errors, isSeq?results[0]:results)
274 				}, err, result)
275 			})
276 		}
277 	}
278 };

281  * microevents.js - https://github.com/jeromeetienne/microevent.js
282 */
283 tQuery.MicroeventMixin	= function(destObj){
284 	destObj.bind	= function(event, fct){
285 		if(this._events === undefined) 	this._events	= {};
286 		this._events[event] = this._events[event]	|| [];
287 		this._events[event].push(fct);
288 		return fct;
289 	};
290 	destObj.unbind	= function(event, fct){
291 		if(this._events === undefined) 	this._events	= {};
292 		if( event in this._events === false  )	return;
293 		this._events[event].splice(this._events[event].indexOf(fct), 1);
294 	};
295 	destObj.trigger	= function(event /* , args... */){
296 		if(this._events === undefined) 	this._events	= {};
297 		if( this._events[event] === undefined )	return;
298 		var tmpArray	= this._events[event].slice(); 
299 		for(var i = 0; i < tmpArray.length; i++){
300 			tmpArray[i].apply(this, Array.prototype.slice.call(arguments, 1))
301 		}
302 	}
303 };
